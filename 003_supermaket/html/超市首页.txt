function menuList(){
	var params= "key=L_${categoryId}_${rgnCode}_json_${categoryId}_1";
	new Ajax.Request('dataEnginerServlet',{
		method:"post",
		parameters: params,
		onSuccess:function(transport){
			var response = transport.responseText;
			var data = eval("("+response+")");
			menus = data;
			showMenu();
			menuEffect();
		},
		onFailure: function(transport){
		}
	});
}

function getDataKey(){
	var request = getRequestParam();
	var key = request['key'];
	if(key==""||key=="undefined"){
		return "";
	}
	return key;
}

function showMenu(){
	var length = menus.length;

function maqFmtCutStr(str,num){
	var strlength = 0;
	for(var i=0;i<str.length;i++){ 
		var intcode=str.charCodeAt(i);
		if(intcode>=0 && intcode<=128) {
			strlength+=1;
		}else{
			strlength+=2;
		}
	}
	if(strlength>num){
		return "<marquee>"+str+"</marquee>";
	}else {
		return str;
	}
}

 }else{

function grabEvent(){
	var kcode = event.which || event.keyCode;
	switch (kcode) {
		case KEY.UP:
		   if (focusPos == "MENU"){
		     focusPos = "SUBMIT";
			 submitBtnEffect();
			 $("menu"+menuObj.focusPos).style.background = "url({respath}/images/tab_hl.png) no-repeat";
		     break;
		   }
		   if (focusPos == "ITEM"){
		    if (contentObj.focusPos >= 5){
			     contentObj.changeList(-5);
				 contentEffect();	
			 }else {
		        focusPos = "MENU";
			     contentEffect();
			 }
			 break;
		   }
		   if (focusPos == "CONFIRM" || focusPos == "BACK"){
		       var num = $("info_num").innerText;
			   if (parseInt(num)>=99){
			     focusPos = "DEL";
				 $("numfocus").style.left = "590px";
			   }else {
				  focusPos = "ADD";
				   $("numfocus").style.left = "715px";
			  }
			      show("numfocus");
				  confirmBtnEffect();
				  backBtnEffect();
			  break;
		   }
		break;
		case KEY.DOWN:
			 if (focusPos == "SUBMIT" || focusPos == "SELECT"){
				 focusPos = "MENU";
				 submitBtnEffect();
				 selectBtnEffect();
				 $("menu"+menuObj.focusPos).style.background = "url({respath}/images/tab--.png) no-repeat";
				 break;
		    }
		    if (focusPos == "MENU"){
			    focusPos = "ITEM";
				contentEffect();
				break;
			}
			if (focusPos == "ITEM"){
			    if(contentObj.focusPos<=5 && contentObj.focusPos+5>=contentObj.dataSize){
					break;
				}
				if(contentObj.focusPos<=5){
					contentObj.changeList(5);
					contentEffect();									
				}	
				break;
			}
			if (focusPos == "ADD" || focusPos == "DEL"){
			   focusPos = "CONFIRM";
			   hide("numfocus");
			   confirmBtnEffect();
			   break;
			}
		break;
		case KEY.LEFT:
			 if (focusPos == "MENU"){
		         menuObj.focusUp();
				 menuEffect();
		        break;
		    }
			if (focusPos == "SELECT"){
			    focusPos = "SUBMIT";
				submitBtnEffect();
			    selectBtnEffect();
				break;
			}
			if (focusPos == "ITEM"){
			    contentObj.focusUp();
				contentEffect();
				break;
			}
			if (focusPos == "BACK"){
				focusPos = "CONFIRM";
				confirmBtnEffect();
				backBtnEffect();
				break;
			}
			if (focusPos == "ADD"){
			   var num = $("info_num").innerText;
			   if (parseInt(num)>1){
			     focusPos = "DEL";
				 $("numfocus").style.left = "590px";
			   }
			    break;
			}
			if (focusPos == "alert_right"){
			   focusPos = "alert_left";
			   alertBtnEffect();
			   break;
			}
		break;
		case KEY.RIGHT:
			 if (focusPos == "MENU"){
		         menuObj.focusDown();
				 menuEffect();
		         break;
		   }
		   if (focusPos == "SUBMIT"){
			    focusPos = "SELECT";
				submitBtnEffect();
			    selectBtnEffect();
				break;
			}
			if (focusPos == "ITEM"){
			    contentObj.focusDown();
				contentEffect();
				break;
			}
			if (focusPos == "CONFIRM"){
			   focusPos = "BACK";
			   confirmBtnEffect();
			   backBtnEffect();
			   break;
			}
			if (focusPos == "DEL"){
			   var num = $("info_num").innerText;
			   if (parseInt(num)<99){
			     focusPos = "ADD";
				 $("numfocus").style.left = "715px";
			   }
			    break;
			}
			if (focusPos == "alert_left"){
			   focusPos = "alert_right";
			   alertBtnEffect();
			   break;
			}
		break;
		case KEY.ENTER:
			if (focusPos == "ITEM"){
			   focusPos = "CONFIRM";
			   $("info_poster").src = "res/content/small/"+contentObj.dataList[contentObj.position].smallpic;
			   $("info_name").innerHTML = contentObj.dataList[contentObj.position].title;
			   $("info_price").innerHTML = contentObj.dataList[contentObj.position].ext1+"Ôª";
			   $("info_num").innerHTML = "1";
			   confirmBtnEffect();
			   backBtnEffect();
			   show("info_alert");
			   break;
			}
			if (focusPos == "ADD"){
			   var num = $("info_num").innerText;
			    $("info_num").innerHTML = parseInt(num)+1;
			   if (parseInt(num) == 98){
			      focusPos = "DEL";
				  $("numfocus").style.left = "590px";
			      $("info_num").style.background = "url({respath}/images/left.png) no-repeat";
			   }else {
			      $("info_num").style.background = "url({respath}/images/middle.png) no-repeat";
			   }
			   break;
			}
			if (focusPos == "DEL"){
			   var num = $("info_num").innerText;
			    $("info_num").innerHTML = parseInt(num)-1;
			   if (parseInt(num) == 2){
			      focusPos = "ADD";
				  $("numfocus").style.left = "715px";
			      $("info_num").style.background = "url({respath}/images/right.png) no-repeat";
			   }else {
			      $("info_num").style.background = "url({respath}/images/middle.png) no-repeat";
			   }
			   break;
			}
			if (focusPos == "CONFIRM"){
			  addShopCart();
			  show("msg_alert");
			  focusPos = "alert_right";
			  alertBtnEffect();
			  closeInfoAlert();
			  break;
			}
			if (focusPos == "BACK" || focusPos == "alert_right"){
				focusPos = "ITEM";
				closeInfoAlert();
				closeMsgAlert();
				break;
			}
			if (focusPos == "alert_left" || focusPos == "SUBMIT"){
			    window.location.href = "dataEnginerServlet?key=L_${categoryId}_${rgnCode}_cart_${categoryId}_1&backurl="+getThisUrl();
				break;
			}
			if (focusPos == "SELECT"){
			    window.location.href = "dataEnginerServlet?key=L_${categoryId}_${rgnCode}_order_${categoryId}_1&backurl="+getThisUrl();
				break;
			}
		break;
		case KEY.PAGEUP:
			prePage();
			break;
		case KEY.PAGEDOWN:
			nextPage();
			break;
		case KEY.BACK:
			if(focusPos == "BACK"||focusPos=="CONFIRM"||focusPos == "alert_right"||focusPos=="alert_left"){
				focusPos = "ITEM";
				closeInfoAlert();
				closeMsgAlert();
				return 0;
			}else{
				toGoBack();
				return 0;
			}
		break;
	}
}

	obj.price = parseFloat(price);
	obj.disNum = parseInt(num);
  
	setCookie("DMX_"+obj.dishId,JSON.stringify(obj));
}
